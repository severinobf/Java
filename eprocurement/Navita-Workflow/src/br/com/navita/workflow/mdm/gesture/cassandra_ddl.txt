CREATE KEYSPACE NAVITA_GESTOS WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 3};

CREATE TABLE NAVITA_GESTOS.GESTOS (
	DESCRICAO_PRODUTO VARCHAR,
	ESTADO_CIVIL INT,
	DATAHORA TIMESTAMP,
	SEXO INT,
	IDADE INT,
	ID_PRODUTO VARCHAR,
	PAPEL VARCHAR,
	DESCRICAO_GESTO VARCHAR
	PRIMARY KEY (ID_PRODUTO, PAPEL, DATAHORA)
);
